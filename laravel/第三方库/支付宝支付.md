<?php

namespace App\Services;

use Alipay\EasySDK\Kernel\Config;
use Alipay\EasySDK\Kernel\Factory;


class AliPayEasyService
{
    public function __construct()
    {
        // 必须在使用前初始化！！！这是你报错的关键
        Factory::setOptions($this->getOptions());
    }

    public function getOptions()
    {
        $options = new Config();
        $options->protocol = 'https';
        //$options->gatewayHost = 'openapi.alipay.com'; //正式请求地址
        $options->gatewayHost = 'openapi-sandbox.dl.alipaydev.com';//沙箱地址 沙箱信息可以查看
        $options->signType = 'RSA2';

        //$options->appId = '<-- 请填写您的AppId，例如：2019022663440152 -->';
        $options->appId = \config('pay.alipay.app_id');

        // 为避免私钥随源码泄露，推荐从文件中读取私钥字符串而不是写入源码中
        //$options->merchantPrivateKey = '<-- 请填写您的应用私钥，例如：MIIEvQIBADANB ... ... -->';
        $options->merchantPrivateKey = \config('pay.alipay.private_key');

//        $options->alipayCertPath = '<-- 请填写您的支付宝公钥证书文件路径，例如：/foo/alipayCertPublicKey_RSA2.crt -->';
//        $options->alipayRootCertPath = '<-- 请填写您的支付宝根证书文件路径，例如：/foo/alipayRootCert.crt" -->';
//        $options->merchantCertPath = '<-- 请填写您的应用公钥证书文件路径，例如：/foo/appCertPublicKey_2019051064521003.crt -->';

        //注：如果采用非证书模式，则无需赋值上面的三个证书路径，改为赋值如下的支付宝公钥字符串即可
        // $options->alipayPublicKey = '<-- 请填写您的支付宝公钥，例如：MIIBIjANBg... -->';
        $options->alipayPublicKey = \config('pay.alipay.ali_public_key');

        //可设置异步通知接收服务地址（可选）
        //$options->notifyUrl = "<-- 请填写您的支付类接口异步通知接收服务地址，例如：https://www.test.com/callback -->";
        $options->notifyUrl = "http://47.107.33.56:1997/api/pay_notice";//必须改为公网地址

        return $options;
    }
}

#  支付宝上线换配置解答

## 我现在需要把laravel项目我的支付宝配置  换成新的一套 但是我每天的订单量有七八万    换的过程中会不会影响业务 比如 无法支付 或者支付了无法成功回调

### 核心结论

**只要按正确流程切换，不会影响任何支付业务**

用户支付、回调、订单状态更新全程无感，不会出现支付失败 / 回调失败 / 丢单。

### 零风险切换步骤（生产环境标准流程）

### 最关键：回调会不会乱？旧订单会不会失效？

✅ 完全不会

我给你讲清楚原理，你就放心了：

1）旧订单的回调

- 用户**之前支付成功**的订单
- 支付宝**仍然使用旧配置回调**
- 回调里会携带旧的 `appid`
- 你的回调接口**可以识别 appid**，自动用旧配置验签

2）新订单的回调

- 新创建的订单
- 使用新配置发起支付
- 回调自动走新配置验签

3）你的回调接口必须兼容「新旧两套配置验签」

这是**生产环境高可用标准写法**：

```php
// 支付宝回调
public function notify(Request $request)
{
    $data = $request->all();
    $appId = $data['app_id']; // 支付宝会返回当前支付的 appid

    // 根据 appid 自动选择对应配置验签
    if ($appId == config('alipay.old.app_id')) {
        $config = config('alipay.old');
    } else {
        $config = config('alipay.new');
    }

    // 验签
    $alipay = new Alipay($config);
    $verify = $alipay->verify($data);

    if (!$verify) {
        return 'fail';
    }

    // 处理订单逻辑...
    return 'success';
}
```

**这个写法 = 新旧回调 100% 兼容，绝对不会出现验签失败、不回调、订单卡住。**

2. 会不会支付成功但回调失败？
