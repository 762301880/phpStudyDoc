#  分期付款数据库结构设计

`installments` 分期信息表

```mysql
-- 创建分期信息表
CREATE TABLE `installments` (
  `id` bigint unsigned NOT NULL AUTO_INCREMENT COMMENT '自增长ID',
  `no` varchar(64) NOT NULL COMMENT '分期流水号',
  `user_id` bigint unsigned NOT NULL COMMENT '用户ID',
  `order_id` bigint unsigned NOT NULL COMMENT '对应的商品订单ID',
  `total_amount` decimal(18,2) NOT NULL DEFAULT 0.00 COMMENT '总本金',
  `count` int unsigned NOT NULL DEFAULT 0 COMMENT '还款期数',
  `fee_rate` float NOT NULL DEFAULT 0 COMMENT '手续费率',
  `fine_rate` float NOT NULL DEFAULT 0 COMMENT '逾期费率',
  `status` varchar(32) NOT NULL DEFAULT '' COMMENT '还款状态',
  PRIMARY KEY (`id`) COMMENT '主键',
  UNIQUE KEY `uk_no` (`no`) COMMENT '分期流水号全局唯一索引',
  KEY `idx_user_id` (`user_id`) COMMENT '用户ID普通索引',
  KEY `idx_order_id` (`order_id`) COMMENT '订单ID普通索引'
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='分期信息表';
```

我们在 installments 表中保存了手续费率，这是因为在运营过程中有可能出现修改手续费率的情况，而已经在还款的分期的手续费率应保持不变，我们在计算手续费的时候应使用这个表中的费率，而不是根据分期期数去读取费率。

接下来是 installment_items 还款计划表：

```mysql
-- 创建还款计划表
CREATE TABLE `installment_items` (
  `id` bigint unsigned NOT NULL AUTO_INCREMENT COMMENT '自增长ID',
  `installment_id` bigint unsigned NOT NULL COMMENT '分期ID',
  `sequence` int unsigned NOT NULL DEFAULT 0 COMMENT '还款顺序编号',
  `base` decimal(18,2) NOT NULL DEFAULT 0.00 COMMENT '当期本金',
  `fee` decimal(18,2) NOT NULL DEFAULT 0.00 COMMENT '当期手续费',
  `fine` decimal(18,2) NOT NULL DEFAULT 0.00 COMMENT '当期逾期费',
  `due_date` datetime DEFAULT NULL COMMENT '还款截止日期',
  `paid_at` datetime DEFAULT NULL COMMENT '还款日期',
  `payment_method` varchar(32) DEFAULT '' COMMENT '还款支付方式',
  `payment_no` varchar(64) DEFAULT '' COMMENT '还款支付平台订单号',
  `refund_status` varchar(32) DEFAULT '' COMMENT '退款状态',
  PRIMARY KEY (`id`) COMMENT '主键',
  KEY `idx_installment_id` (`installment_id`) COMMENT '分期ID索引',
  KEY `idx_due_date` (`due_date`) COMMENT '还款日期索引（便于查询逾期）'
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='还款计划表';
```
