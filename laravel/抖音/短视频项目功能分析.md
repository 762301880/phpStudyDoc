##  任务金币功能

```php
│   ├── 播放返金币系统
│   │   ├── 打开App播放 +1金币（每日5次）
│   │   ├── 每5分钟 +2金币
│   │   ├── 累计30分钟 +10金币 + 分享弹窗
│   │   ├── 分享作品 +5金币（每日3次）
│   │   └── 裂变：双方各+20金币
```

### 核心讲解一下看X分钟领取金币功能(如何防作弊)

#### 完整防作弊风控规则说明

1. **签名校验**：参数 + token + 秘钥 MD5，抓包改任何字段都会签名失败
2. **时间戳校验**：前端本地时间和服务器相差超过 30 秒直接判定作弊，解决改系统时间刷时长
3. **分片上报 + 单次上限**：每 15s 上报，单次最多 20 秒，杜绝一次性传几千秒假数据
4. **Redis 限流**：1 分钟最多 10 次上报，拦截脚本无限循环刷接口
5. **后台 / 锁屏不计时长**：页面不可见时上报时长直接丢弃，防止息屏挂机
6. **单视频单日上限**：单视频最多看 10 分钟，防止循环重复刷同一条视频
7. **日志全留存**：所有上报行为入库，作弊用户可后台筛选封禁
8. **设备指纹**：携带 device_id，同一设备多账号刷时长可关联风控

#### 拓展进阶风控（可选）

1. 新增滑动行为检测：前端上报滑动切换视频时间，后端判断短时间极速滑动视为脚本，当日时长减半
2. 增加 IP 风控：同一 IP 大量账号同时刷视频触发预警
3. 夜间时长折扣：0-6 点观看时长乘以 0.5 系数
4. 作弊用户标记 redis 黑名单，拦截所有时长上报请求
5. 定时任务清理过期 Redis 缓存，优化性能



### 后端逻辑参考

```php
<?php

namespace app\api\controller;

use app\common\controller\Api;
use app\common\service\RedisLock;
use app\common\service\RedisService;
use app\common\service\UserGoldLogService;
use app\common\service\UserGoldService;
use app\common\service\UserPlayTaskService;
use app\common\traits\UserInitTrait;
use think\Db;
use think\Exception;
use think\Request;

class UserPlayTask extends Api
{
    protected $noNeedLogin = [];
    protected $noNeedRight = ['*'];
    use UserInitTrait;

    protected $userPlayTaskService = null;

    protected $lock = null;
    protected $redis = null;

    public function __construct(Request $request = null)
    {
        parent::__construct($request);
        $this->userPlayTaskService = new UserPlayTaskService();
        $this->initUserInfo();
        $this->lock = new RedisLock();
        $this->redis = RedisService::getInstance();
    }

    /**
     * 打开App播放奖励领取次数
     * @return void
     */
    public function openAppGold(Request $request)
    {
        $lockKey = "openAppGold:lock:{$this->user_id}";
        if (!$this->lock->lock($lockKey, 10)) $this->error('操作频繁请稍后');
        $addGoldNum = 1;
        try {
            Db::startTrans();
            $task = $this->userPlayTaskService->getMyTodayPlayTask();
            if (!empty($task->open_app_count) && $task->open_app_count >= 5) $this->error('到达播放投放5次金币最大次数');
            $task->open_app_count++;
            $task->save();
            //记录金币日志
            $userGoldInsertData = [
                'gold_num' => $addGoldNum,
                'rel_id' => $task->id ?? "",
                'remark' => '今日打开App播放奖励领取'
            ];
            (new UserGoldLogService())->saveLog($userGoldInsertData);

            (new UserGoldService())->create($addGoldNum);

        } catch (Exception $exception) {
            Db::rollback();
            $this->error("打开获取金币失败", $exception->getMessage() . $exception->getLine());
        } finally {
            // 2. 释放锁（一定执行）
            $this->lock->unlock();
        }
        Db::commit();
        $this->success("打开获取金币成功", ['gold' => $addGoldNum]);
    }

    /**
     * 每日分享加金币
     * 分享作品 +5金币（每日3次）
     * @param Request $request
     * @return void
     */
    public function shareAppGold(Request $request)
    {
        $lockKey = "shareAppGold:lock:{$this->user_id}";
        if (!$this->lock->lock($lockKey, 10)) $this->error('操作频繁请稍后');
        $addGoldNum = 5;//分享添加金币数量

        Db::startTrans();
        try {
            $task = $this->userPlayTaskService->getMyTodayPlayTask();
            if (!empty($task->share_count) && $task->share_count >= 3) $this->error('每日分享次数最多3次');
            $task->share_count++;//分享次数+1
            $task->save();
            //记录金币日志
            $userGoldInsertData = [
                'gold_num' => $addGoldNum,
                'rel_id' => $task->id ?? "",
                'remark' => '今日分享'
            ];
            (new UserGoldLogService())->saveLog($userGoldInsertData);

            (new UserGoldService())->create($addGoldNum);
        } catch (Exception $exception) {
            Db::rollback();
        } finally {
            $this->lock->unlock();
        }
        Db::commit();
        $this->success("分享成功", [
            'gold' => $addGoldNum
        ]);

    }


    /**
     * 发放30分钟累计奖励10金币
     */
    protected function addThirtyMinReward($task)
    {
        $task->thirty_min_reward_status = 1;
        $task->save();
        $logData = [
            'gold_num' => 10,
            'rel_id' => $task->id,
            'remark' => '当日累计观看30分钟奖励'
        ];
        (new UserGoldLogService())->saveLog($logData);
        (new UserGoldService())->create(10);
    }

    /**
     * 播放心跳
     * 每15秒调用一次
     */
    public function playHeartbeat(Request $request)
    {
        $userId = $this->user_id;
        // 每日播放总时长Redis key，当日过期
        $totalSecKey = "user_play_total_sec:{$userId}";
        // 自然日过期：剩余秒数到今天24点
        $now = time();
        $todayEnd = strtotime(date('Y-m-d 23:59:59'));
        $ttlDay = $todayEnd - $now;

        // 原子累加15秒（单条命令无并发脏读）
        $currentTotal = $this->redis->incrBy($totalSecKey, 15);
        // 首次初始化过期时间
        if ($this->redis->ttl($totalSecKey) < 0) {
            $this->redis->expire($totalSecKey, $ttlDay);
        }

        // 核心风控：超过30分钟直接拦截，不查DB、不进事务
        if ($currentTotal >= 1800) {
            $this->success("今日已达30分钟观看奖励上限，次日再来");
        }

        $lockKey = "playHeartbeat:lock:{$userId}";
        if (!$this->lock->lock($lockKey, 5)) {
            $this->error("操作频繁");
        }

        Db::startTrans();
        try {
            $task = $this->userPlayTaskService->getMyTodayPlayTask();

            $beforePlayHeartBeatTime = $task->play_heart_beat_time ?? "";
            $diff = (strtotime(date("Y-m-d H:i:s")) - strtotime($beforePlayHeartBeatTime)) < 15;
            if (!empty($beforePlayHeartBeatTime) && $diff) $this->error("非法操作");

            $this->checkHeartbeatLimit();

            // 这里不再手动+15，Redis已经加完，DB同步Redis最新值
            $task->total_play_min = $currentTotal;
            $task->play_heart_beat_time = date("Y-m-d H:i:s");

            $gold = 0;

            // 5分钟奖励逻辑不变，读取task里同步后的总时长
            $canRewardFive = floor($task->total_play_min / 300) - $task->five_min_reward_times;
            if ($canRewardFive > 0) {
                $task->five_min_reward_times += $canRewardFive;
                $gold += $canRewardFive * 2;
                // 循环生成日志+发金币（后面优化建议批量插入）
                for ($i = 0; $i < $canRewardFive; $i++) {
                    (new UserGoldLogService())->saveLog([
                        'gold_num' => 2,
                        'rel_id' => $task->id,
                        'remark' => '观看5分钟奖励'
                    ]);
                    (new UserGoldService())->create(2);
                }
            }

            // 30分钟一次性奖励
            if ($task->total_play_min >= 1800 && $task->thirty_min_reward_status == 0) {
                $this->addThirtyMinReward($task);
            }

            $task->save();
            Db::commit();

            $this->success("播放心跳成功", [
                'add_gold' => $gold,
                'total_play_second' => $currentTotal,
            ]);
        } catch (Exception $e) {
            Db::rollback();
            // DB失败回滚，同步撤销Redis增量（关键兜底，防止数据不一致）
            $this->redis->decr($totalSecKey, 15);
            $this->error("播放心跳失败", $e->getMessage());
        } finally {
            $this->lock->unlock();
        }
    }
    public function checkHeartbeatLimit()
    {
        $key = "play_heartbeat_limit:{$this->user_id}";

        $count = $this->redis->get($key);

        if ($count >= 5) {
            $this->error("心跳频繁，请稍后");
        }
        if (!$count) {
            $this->redis->set($key, 1, 'EX', 60);
        } else {
            // 后续+1
            $this->redis->incr($key);
        }
    }


}
```
