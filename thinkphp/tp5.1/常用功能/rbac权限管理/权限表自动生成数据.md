## 先看一下表结构

```sql
CREATE TABLE `permission` (
  `id` bigint unsigned NOT NULL AUTO_INCREMENT,
  `name` varchar(255) COLLATE utf8mb4_unicode_ci NOT NULL COMMENT '权限标题',
  `p_id` int NOT NULL DEFAULT '0' COMMENT '父级id默认0',
  `controller` varchar(255) COLLATE utf8mb4_unicode_ci NOT NULL COMMENT '控制器名称',
  `action` varchar(255) COLLATE utf8mb4_unicode_ci NOT NULL COMMENT '方法名称',
  `param` json DEFAULT NULL COMMENT '参数', --此字段不写参数预备使用
  `api` varchar(255) COLLATE utf8mb4_unicode_ci NOT NULL DEFAULT '' COMMENT '请求api',
  `level` tinyint NOT NULL DEFAULT '1' COMMENT '等级',
  `sort` tinyint NOT NULL DEFAULT '100' COMMENT '默认排序',
  `created_at` timestamp NULL DEFAULT NULL,
  `updated_at` timestamp NULL DEFAULT NULL,
  PRIMARY KEY (`id`)
) ENGINE=InnoDB AUTO_INCREMENT=17 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;
```

## 说明

### 1. 行业通用三级树形权限结构（90% 后台 / 中台都是这套）

```php
一级：模块（level=1，p_id=0）→ 用户、权限管理、系统设置
二级：菜单（level=2，p_id=一级ID）→ 用户列表、角色列表、用户角色
三级：接口操作（level=3，p_id=二级ID）→ 列表、新增、编辑、删除
```

### 字段分工（企业统一规范）

| 层级     | controller / action | api / param          | 作用                         |
| -------- | ------------------- | -------------------- | ---------------------------- |
| 一级模块 | 空字符串            | 空 / NULL            | 仅做分类，无接口、无控制器   |
| 二级菜单 | 空字符串            | 空 / NULL            | 页面路由节点，不绑定后端接口 |
| 三级操作 | 填控制器 + 方法     | 填接口地址、请求方式 | 真实后端接口权限控制点       |

### 命令生成结构权限

```php
<?php

namespace App\Console\Commands;

use App\Models\Permission;
use Illuminate\Console\Command;
use Illuminate\Routing\Router;
use Illuminate\Support\Facades\DB;

/**
 * CREATE TABLE `permission` (
 * `id` bigint unsigned NOT NULL AUTO_INCREMENT,
 * `name` varchar(255) COLLATE utf8mb4_unicode_ci NOT NULL COMMENT '权限标题',
 * `p_id` int NOT NULL DEFAULT '0' COMMENT '父级id默认0',
 * `controller` varchar(255) COLLATE utf8mb4_unicode_ci NOT NULL COMMENT '控制器名称',
 * `action` varchar(255) COLLATE utf8mb4_unicode_ci NOT NULL COMMENT '方法名称',
 * `param` json DEFAULT NULL COMMENT '参数',
 * `api` varchar(255) COLLATE utf8mb4_unicode_ci NOT NULL DEFAULT '' COMMENT '请求api',
 * `level` tinyint NOT NULL DEFAULT '1' COMMENT '等级',
 * `sort` tinyint NOT NULL DEFAULT '100' COMMENT '默认排序',
 * `created_at` timestamp NULL DEFAULT NULL,
 * `updated_at` timestamp NULL DEFAULT NULL,
 * PRIMARY KEY (`id`)
 * ) ENGINE=InnoDB AUTO_INCREMENT=17 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;
 */
class PermissionGenerate extends Command
{
    protected $signature = 'permission:generate';
    protected $description = '权限自动生成(API作为唯一标识，改名自动更新，废弃自动删除)';

    protected $router;
    // 缓存一级/二级菜单ID：key=name_pid
    protected array $menuCache = [];
    // 本次脚本所有有效的API地址集合，用于清理废弃权限
    protected array $validApiList = [];

    public function __construct(Router $router)
    {
        parent::__construct();
        $this->router = $router;
    }

    public function handle()
    {
        $routes = $this->router->getRoutes();
        $this->menuCache = [];
        $this->validApiList = [];

        DB::beginTransaction();
        try {
            foreach ($routes as $route) {
                $methods = $route->methods();
                $uri = $route->uri();
                $routeName = $route->getName();
                $action = $route->getAction();
                $defaults = $route->defaults;

                //只处理带自定义权限名称的后台路由
                if (empty($defaults['name']) || !str_starts_with($uri, 'api/admin')) {
                    continue;
                }

                if (empty($defaults['name'])) continue;

                $nameArr = explode('/', $defaults['name']);
                if (count($nameArr) !== 3) {
                    $this->warn("路由 {$uri} name格式错误，要求：模块/菜单/操作");
                    continue;
                }
                [$level1Name, $level2Name, $level3Name] = $nameArr;

                // 1. 创建一级模块 level=1 p_id=0
                $pid1 = $this->getOrCreateMenu([
                    'name' => $level1Name,
                    'p_id' => 0,
                    'level' => 1,
                    'controller' => '',
                    'action' => '',
                    'api' => '',
                    'param' => null,
                ]);

                // 2. 创建二级菜单 level=2
                $pid2 = $this->getOrCreateMenu([
                    'name' => $level2Name,
                    'p_id' => $pid1,
                    'level' => 2,
                    'controller' => '',
                    'action' => '',
                    'api' => '',
                    'param' => null,
                ]);

                // 解析控制器、方法
                [$controller, $actionName] = $this->parseControllerAction($action);

                $paramData = [
                    'uri' => $uri,
                    'methods' => $methods,
                    'route_name' => $routeName
                ];

                // 3. 三级权限：以api为唯一匹配键，存在则全量更新
                $this->updateThirdByApi([
                    'match_api' => $uri,
                    'data' => [
                        'name' => $level3Name,
                        'p_id' => $pid2,
                        'level' => 3,
                        'controller' => $controller,
                        'action' => $actionName,
                        'param' => null,
                        'sort' => 100,
                    ]
                ]);
                // 存入有效API清单
                $this->validApiList[] = $uri;

                $this->info("已同步接口：{$defaults['name']} [{$uri}]");
            }

            // 删除废弃三级权限：level=3 且 api不在本次有效列表
            $delCount = Permission::where('level', 3)
                ->whereNotIn('api', $this->validApiList)
                ->delete();
            if ($delCount > 0) {
                $this->warn("清理废弃接口权限共 {$delCount} 条");
            }

            DB::commit();
            $this->info('✅ 权限同步完成，已删除失效接口权限');
        } catch (\Exception $e) {
            DB::rollBack();
            $this->error("❌ 执行失败：" . $e->getMessage() . ':' . $e->getLine());
            $this->error($e->getTraceAsString());
        }
    }

    /**
     * 一级、二级菜单：仅新增，不覆盖原有数据（保护手动排序、修改）
     */
    protected function getOrCreateMenu(array $data): int
    {
        $cacheKey = $data['name'] . '_' . $data['p_id'];
        if (isset($this->menuCache[$cacheKey])) {
            return $this->menuCache[$cacheKey];
        }

        $perm = Permission::firstOrCreate(
            ['name' => $data['name'], 'p_id' => $data['p_id']],
            array_merge(['sort' => 100], $data)
        );
        $this->menuCache[$cacheKey] = $perm->id;
        return $perm->id;
    }

    /**
     * 三级接口权限：按api匹配，改名/换控制器都会更新原有记录
     * @param array $params ['match_api' => 接口地址, 'data' => 待更新字段]
     */
    protected function updateThirdByApi(array $params): void
    {
        $api = $params['match_api'];
        $saveData = $params['data'];

        Permission::updateOrCreate(
        // 匹配条件：唯一接口地址
            ['api' => $api, 'level' => 3],
            $saveData
        );
    }

    /**
     * 拆分控制器与方法名
     */
    protected function parseControllerAction(array $action): array
    {
        if (!isset($action['controller'])) {
            return ['closure', ''];
        }
        $controllerStr = $action['controller'];
        [$fullCtrl, $act] = explode('@', $controllerStr);
        return [class_basename($fullCtrl), $act];
    }
}
```
