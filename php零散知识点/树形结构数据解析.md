## 示例数据表

```sql
CREATE TABLE `categories` (
  `id` bigint unsigned NOT NULL AUTO_INCREMENT COMMENT '分类ID',
  `name` varchar(255) COLLATE utf8mb4_unicode_ci NOT NULL COMMENT '分类名称',
  `pid` bigint unsigned NOT NULL DEFAULT 0 COMMENT '父级ID，0=一级分类',
  `status` tinyint NOT NULL DEFAULT 1 COMMENT '状态 0=禁用 1=启用',
  `is_directory` tinyint NOT NULL DEFAULT 0 COMMENT '是否为父目录 0=否 1=是',
  `level` tinyint NOT NULL DEFAULT 1 COMMENT '分类层级 1一级 2二级 3三级',
  `created_at` timestamp NULL DEFAULT NULL,
  `updated_at` timestamp NULL DEFAULT NULL,
  PRIMARY KEY (`id`),
  KEY `idx_pid` (`pid`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci COMMENT='商品分类表';
```

###  php递归循环解析树形结构(推荐性能好)

```php
<?php

namespace App\Http\Controllers;

use App\Traits\ApplyResponseLayout;
use Illuminate\Http\Request;

class TestController extends Controller
{
    use ApplyResponseLayout;

    public function test(Request $request)
    {
        $res = \DB::table('categories')
            ->orderBy('level')
            ->orderBy('id')
            ->get();

        // ✅ 关键：这样转才是 纯PHP数组
        $categoryList = json_decode(json_encode($res), true);

        // ✅ 生成树
        $tree = $this->buildTree($categoryList);

        return $tree;
    }

    /**
     * 递归生成无限极分类树
     */
    public function buildTree(array $data, int $pid = 0): array
    {
        $tree = [];
        // 1. 遍历所有数据
        foreach ($data as $item) {
             // 2. 如果这个条目的爸爸 是 当前找的爸爸
            if ($item['pid'] == $pid) {
                // ✅ 必须加 $this->
                // 3. 递归：继续找这个条目的儿子
                $children = $this->buildTree($data, $item['id']);
                // 4. 把儿子放进这个条目里
                if (!empty($children)) {
                    $item['children'] = $children;
                }

                $tree[] = $item;
            }
        }
        return $tree;
    }
}
```
