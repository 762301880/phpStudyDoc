### Comment.php

```php
<?php

namespace app\api\controller\video;

use app\common\controller\Api;
use app\common\model\VideoComment;
use app\common\service\UserService;
use app\common\service\VideoCommentService;
use app\common\traits\UserInitTrait;
use app\common\validate\VideoCommentValidate;
use think\Db;
use think\Exception;
use think\Request;

class Comment extends Api
{
    use UserInitTrait;

    protected $noNeedLogin = [];
    protected $noNeedRight = '*';
    protected $videoCommentService = null;

    public function __construct(Request $request = null)
    {
        parent::__construct($request);
        $this->videoCommentService = new VideoCommentService();
        $this->initUserInfo();
    }

    public function add(Request $request)
    {
        try {
            $params = $request->param();
            $validate = new VideoCommentValidate();
            if (!$validate->check($params)) $this->error($validate->getError());
            $bool = $this->videoCommentService->addComment($params);
            if (!$bool) $this->error("评论失败");
        } catch (Exception $exception) {
            $this->error($exception->getMessage());
        }
        $this->success("评论成功");
    }

    public function getCommentList(Request $request)
    {
        $drama_chapter_id = $request->param('drama_chapter_id', '');
        if (empty($drama_chapter_id)) $this->error('短剧剧集id不能为空');
        $topComments = VideoComment::alias('comment')
            ->field('comment.*')
            ->where('comment.parent_id', 0)
            ->where('comment.drama_chapter_id', $drama_chapter_id)
            ->view('video_comment_like like', 'status', "like.comment_id=comment.id and like.user_id=$this->user_id", 'LEFT')
            ->order('comment.create_time', 'desc')
            ->paginate(10);


        $topIds = array_column($topComments->items(), 'id');
        // 内层子查询
        $sub = Db::name('video_comment')
            ->alias('comment')
            ->fieldRaw('comment.*, ROW_NUMBER() OVER (PARTITION BY root_id ORDER BY id ASC) as rn')
            ->where('root_id', 'in', $topIds)
            ->view('video_comment_like like', 'status', "like.comment_id=comment.id and like.user_id=$this->user_id", 'LEFT')
            ->whereRaw('comment.id != root_id')
            ->buildSql(); //生成括号子查询SQL字符串


        // 外层筛选rn=1
        $firstReplies = Db::table($sub . ' t')
            ->where('t.rn', 1)
            ->select();


        $firstReplyMap = [];
        foreach ($firstReplies as $item) {
            $firstReplyMap[$item['root_id']] = $item;
        }

        $replyCounts = VideoComment::whereIn('root_id', $topIds)
            ->field('root_id, COUNT(*) as total')
            ->group('root_id')
            ->column('COUNT(*) as total', 'root_id');

        $result = [];

        $userIdMapNickName = UserService::getIdMapNickname();

        foreach ($topComments->items() as $comment) {

            $firstReply = $firstReplyMap[$comment['id']] ?? null;

            $total = $replyCounts[$comment['id']] ?? 1;

            // 已展示：一级 + 子评论
            $shown = $firstReply ? 2 : 1;

            $remain = max(0, $total - $shown);

            $result[] = [
                'id' => $comment['id'] ?? "",
                'content' => $comment['content'] ?? "",
                'like_count' => $comment['like_count'] ?? "",
                'is_lick' => $comment['status'] ?? 0,

                // 👉 只展示1条子评论
                'reply' => $firstReply ? [
                    'id' => $firstReply['id'],
                    'content' => $firstReply['content'] ?? "",
                    'user_id' => $firstReply['user_id'] ?? "",
                    'user_name' => $userIdMapNickName[$firstReply['user_id']] ?? "",
                    'reply_to_user_id' => $firstReply['reply_to_user_id'] ?? "",
                    'reply_to_user_name' => $userIdMapNickName[$firstReply['reply_to_user_id']] ?? "",
                    'like_count' => $firstReply['like_count'] ?? "",
                    'is_lick' => $firstReply['status'] ?? 0,
                ] : null,

                // 👉 剩余条数（关键）
                'reply_count' => $remain,

                'has_more' => $remain > 0,
            ];
        }
        $this->success('评论返回成功', $result);
    }


    function getRemainReplies(Request $request)
    {
        $rootId = $request->param('root_id');
        $page = $request->param('page', 1);
        $pageSize = $request->param('pageSize', 10);
        // 👉 1. 自动查“首条子评论”（避免依赖前端）
        $firstReply = VideoComment::where('root_id', $rootId)
            ->where('id', '<>', $rootId)
            ->order('create_time', 'asc')
            ->find();

        // 👉 2. 构造排除ID
        $excludeIds = [$rootId];

        if ($firstReply) {
            $excludeIds[] = $firstReply->id;
        }

        // 👉 3. 查询剩余评论（稳定分页🔥）
        $list = VideoComment::alias('comment')
            ->field('comment.*,ifnull(like.status,0) as is_like')
            ->where('comment.root_id', $rootId)
            ->whereNotIn('comment.id', $excludeIds)
            ->view('video_comment_like like', 'status', "like.comment_id=comment.id and like.user_id=$this->user_id", 'LEFT')
            ->order('comment.create_time', 'asc')
            ->order('comment.id', 'asc') // 👉 防止时间相同导致分页错乱
            ->paginate($pageSize);
        $this->success('剩余数据返回成功', $list);
    }

    public function like(Request $request)
    {
        $id = $request->param('id');
        if (empty($id)) $this->error('主键不能为空');

    }

    public function unlike(Request $request)
    {
        $id = $request->param('id');
        if (empty($id)) $this->error('主键不能为空');
    }
}
```

### VideoCommentService

```php
<?php

namespace app\common\service;

use app\common\model\VideoComment as VideoCommentModel;
use app\common\traits\UserInitTrait;
use think\Db;
use think\Exception;

class VideoCommentService
{
    use UserInitTrait;

    public function __construct()
    {
        $this->initUserInfo();
    }

    public function addComment($data)
    {
        $content = $data['content'] ?? "";//评论内容
        $drama_chapter_id = $data['drama_chapter_id'] ?? "";//评论内容
        $parent_id = $data['parent_id'] ?? 0;//父级id
        try {
            Db::startTrans();
            $model = new VideoCommentModel();
            $model->user_id = $this->user_id;
            $model->content = $content;
            $model->parent_id = $parent_id;
            $model->drama_chapter_id = $drama_chapter_id;
            $model->reply_to_comment_id = $parent_id;
            $model->reply_to_user_id = self::getReplyToUserId($parent_id);
            $model->save();
            self::saveRootId($model);
        } catch (Exception $exception) {
            Db::rollback();
            throw new Exception($exception->getMessage());
        }
        Db::commit();
        return true;
    }

    protected static function getReplyToUserId(int $parent_id)
    {
        if ($parent_id === 0) return 0; //自己评论没有回复返回0
        return VideoCommentModel::find($parent_id)->user_id ?? "";
    }

    /**
     * 递归获取顶级评论 ID（多级评论也能精准找到 root_id）
     */
    protected static function saveRootId(VideoCommentModel $model): bool
    {
        if ($model->parent_id == 0) {
            $model->root_id = $model->id;
            return $model->save();
        }
        //不为0直接返回父类的root_id即可 因为父类的root_id都是一样的
        $model->root_id = $model->parent_id;
        return $model->save();

    }
}
```
